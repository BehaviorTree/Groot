CMAKE_MINIMUM_REQUIRED(VERSION 3.1)

PROJECT(GrootTest LANGUAGES CXX)

find_package(Qt5Test REQUIRED)
find_package(Qt5Gui REQUIRED)

SET(CMAKE_AUTOMOC ON)
SET(CMAKE_AUTORCC ON)
SET(CMAKE_INCLUDE_CURRENT_DIR ON)
SET(CMAKE_CXX_STANDARD 11)
SET(CMAKE_CXX_STANDARD_REQUIRED ON)
ENABLE_TESTING()

set(RESOURCE_FILES
    ../QtNodeEditor/resources/resources.qrc
    ../bt_editor/resources/icons.qrc
    ../test_data/test_files.qrc
    )

add_executable(${PROJECT_NAME}
    groot_test_base.cpp
    groot_test.cpp
    ${RESOURCE_FILES} )

target_link_libraries(${PROJECT_NAME} PRIVATE Qt5::Gui Qt5::Test behavior_tree_editor)
add_test(${PROJECT_NAME} COMMAND ${PROJECT_NAME})
